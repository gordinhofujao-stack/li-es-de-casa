//Exercício 5: 

 


public class Exercicio5 { 

    public static void main(String[] args) { 

        Scanner scanner = new Scanner(System.in); 

 


        System.out.print("Digite o primeiro número: "); 

        double numero1 = scanner.nextDouble(); 

 

        System.out.print("Digite o segundo número: "); 

        double numero2 = scanner.nextDouble(); 

 

        System.out.print("Digite a opção desejada (M - Média, S - Diferença, P - Produto, D - Divisão): "); 

        char opcao = scanner.next().toUpperCase().charAt(0); 

 

        double resultado; 

 


        switch (opcao) { 

            case 'M': 

                resultado = (numero1 + numero2) / 2; 

                System.out.println("Média entre os números: " + resultado); 

                break; 

            case 'S': 

                resultado = Math.abs(numero1 - numero2); // diferença do maior pelo menor 

                System.out.println("Diferença do maior pelo menor: " + resultado); 

                break; 

            case 'P': 

                resultado = numero1 * numero2; 

                System.out.println("Produto dos números: " + resultado); 

                break; 

            case 'D': 

                if (numero2 != 0) { 

                    resultado = numero1 / numero2; 

                    System.out.println("Divisão do primeiro pelo segundo: " + resultado); 

                } else { 

                    System.out.println("Impossível dividir por zero!"); 

                } 

                break; 

            default: 

                System.out.println("Opção inválida! Digite M, S, P ou D."); 

                break; 

        } 

 

        scanner.close(); 

    } 

}
