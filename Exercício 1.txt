
public class Exercicio1 { 

    public static void main(String[] args) { 

        int numeroDoDia = 1; // Suponha que este é o número lido do usuário 

 

        String dia; 

 

        switch (numeroDoDia) { 

            case 1: 

                dia = "Domingo"; 

                break; 

            case 2: 

                dia = "Segunda-feira"; 

                break; 

            case 3: 

                dia = "Terça-feira"; 

                break; 

            case 4: 

                dia = "Quarta-feira"; 

                break; 

            case 5: 

                dia = "Quinta-feira"; 

                break; 

            case 6: 

                dia = "Sexta-feira"; 

                break; 

            case 7: 

                dia = "Sábado"; 

                break; 

            default: 

                dia = "Número inválido. Digite um número de 1 a 7."; 

                break; 

        } 

 

        System.out.println(dia); 

    } 

}
