import java.util.Scanner;

public class exercicio3 {
    
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("Digite o numero:");
        int num = scanner.nextInt();

        int n = 1;
        
        while (n <= num) {
            System.out.println(n);
            n++;
        }

        scanner.close();
    }
}
